{{- /* Post Badge Partial - NEW: posts published within 48 hours - HOT: posts published 2-7 days ago (simulates trending until real analytics connected) - Can also be overridden with frontmatter: hot: true or featured: true */ -}} {{- $now := now }} {{- $postAge := $now.Sub .Date }} {{- $hoursOld := div $postAge.Hours 1 }} {{- /* Check frontmatter overrides */ -}} {{- $isHot := .Params.hot | default false }} {{- $isFeatured := .Params.featured | default false }} {{- /* Auto-calculate based on age */ -}} {{- $isNew := lt $hoursOld 48 }} {{- $isTrending := and (ge $hoursOld 48) (lt $hoursOld 168) }} {{- /* Render badge */ -}} {{- if or $isFeatured }} ★ FEATURED {{- else if $isNew }} NEW {{- else if or $isHot $isTrending }} 🔥 HOT {{- end -}}
